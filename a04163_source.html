440int check_semantics(size_t in_count, arax_data **input, size_t out_count,
441 arax_data **output)
442{
443 size_t io_cnt;
444 size_t dup_cnt;
445 size_t all_io = out_count + in_count;
446 arax_data_s *temp_data_1 = 0;
447 arax_data_s *temp_data_2 = 0;
448
449 for (io_cnt = 0; io_cnt < all_io; io_cnt++) {
450 // Choose from input or output
451 if (io_cnt < in_count)
452 temp_data_1 = input[io_cnt];
453 else
454 temp_data_1 = output[io_cnt - in_count];
455 // check Validity temp_data_1
456 if (!temp_data_1) {
457 fprintf(stderr, "NULL input #%lu\n", io_cnt);
458 return 0;
459 }
460 if (temp_data_1->obj.type != ARAX_TYPE_DATA) {
461 fprintf(stderr, "Input #%lu not valid data\n", io_cnt);
462 return 0;
463 }
464 // Check duplicates
465 for (dup_cnt = 0; dup_cnt < all_io; dup_cnt++) {
466 // Choose from input or output
467 if (dup_cnt < in_count)
468 temp_data_2 = input[dup_cnt];
469 else
470 temp_data_2 = output[dup_cnt - in_count];
471 // check Validity temp_data_2
472 if (!temp_data_2) {
473 fprintf(stderr, "NULL input #%lu\n", dup_cnt);
474 return 0;
475 }
476 if (temp_data_2->obj.type != ARAX_TYPE_DATA) {
477 fprintf(stderr, "Input #%lu not valid data\n", dup_cnt);
478 return 0;
479 }
480 }
481 }
482 return 1;
483} /* check_semantics */
