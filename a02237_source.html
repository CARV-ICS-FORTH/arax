133void arax_data_migrate_accel(arax_data_s *data, arax_accel *accel)
134{
135 arax_assert_obj(data, ARAX_TYPE_DATA);
136 arax_assert(accel); // Must be given a valid accelerator
137
138 if (data->accel == accel) // Already assigned to accel - nop
139 return;
140
141 if (!data->accel) { // Unassinged data - assign and increase accel
142 arax_assert(!data->remote); // How can you have a remote ptr, with no accel?
143 arax_object_ref_inc(((arax_object_s *) accel));
144 data->accel = accel;
145 return;
146 }
147
148 arax_object_type_e data_accel_type = ((arax_object_s *) (data->accel))->type;
149 arax_object_type_e accel_accel_type = ((arax_object_s *) (accel))->type;
150
151 switch (TYPE_MASK(data_accel_type, accel_accel_type) ) {
152 case TYPE_MASK(ARAX_TYPE_VIRT_ACCEL, ARAX_TYPE_VIRT_ACCEL): {
153 arax_vaccel_s *dvac = (arax_vaccel_s *) (data->accel);
154 arax_vaccel_s *avac = (arax_vaccel_s *) (accel);
155 if (dvac->phys == avac->phys) { // Both use the same physical accel - just migrate ref counts
156 arax_object_ref_dec(((arax_object_s *) dvac));
157 arax_object_ref_inc(((arax_object_s *) avac));
158 data->accel = accel;
159 return;
160 } else { // Device 2 Device migration - migrate data and ref counts
161 fprintf(stderr, "Different device data migration!\n");
162 arax_assert(!"arax_data_migrate_accel: D2D not implemented!");
163 }
164 }
165 break;
166 default: {
167 fprintf(stderr, "%s():Data migration not implemented(%s:%s,%s:%s)!\n",
168 __func__,
169 arax_object_type_to_str(data_accel_type), ((arax_object_s *) (data->accel))->name,
170 arax_object_type_to_str(accel_accel_type), ((arax_object_s *) (accel))->name
171 );
172 arax_assert(!"No migration possible");
173 }
174 break;
175 }
176} /* arax_data_migrate_accel */
